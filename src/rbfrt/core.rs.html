20use serde::Deserialize;
21#[derive(Deserialize, Debug)]
22pub(crate) struct Configuration {
23    pub(crate) p4_devices: Vec<DeviceConfig>,
24}
25
26#[derive(Deserialize, Debug)]
27pub(crate) struct DeviceConfig {
28    #[serde(alias = "device-id")]
29    #[allow(dead_code)]
30    pub(crate) device_id: u32,
31    pub(crate) p4_programs: Vec<P4Program>,
32}
33
34#[derive(Deserialize, Debug)]
35pub(crate) struct P4Program {
36    #[serde(alias = "program-name")]
37    pub(crate) program_name: String,
38    #[serde(alias = "bfrt-config")]
39    pub(crate) bfrt_config: String,
40    pub(crate) p4_pipelines: Vec<P4Pipeline>,
41}
42
43#[derive(Deserialize, Debug)]
44pub(crate) struct P4Pipeline {
45    pub(crate) p4_pipeline_name: String,
46    pub(crate) context: String,
47    pub(crate) config: String,
48    pub(crate) pipe_scope: Vec<u32>,
49}
