20use crate::table::ToBytes;
21
22/// Represents data associated with an action.
23///
24/// The data represents a single value, e.g., `string` or `bool`.
25/// For further information, see [Open-Tofino protobuf definition](https://github.com/barefootnetworks/Open-Tofino/blob/master/share/bf_rt_shared/proto/bfruntime.proto).
26#[derive(Debug, Clone)]
27pub struct ActionData {
28    /// Name of the action parameter
29    key: String,
30    /// Data of the action parameter.
31    /// The value is converted to a byte vector.
32    data: Vec<u8>,
33}
34
35/// Represents data associated with an action.
36///
37/// The data represents a list of values, e.g., `IntArray` or `BoolArray`.
38/// For further information, see [Open-Tofino protobuf definition](https://github.com/barefootnetworks/Open-Tofino/blob/master/share/bf_rt_shared/proto/bfruntime.proto).
39#[derive(Debug, Clone)]
40pub struct ActionDataRepeated {
41    /// Name of the action parameter.
42    key: String,
43    /// List of data of the action parameter.
44    /// Each value is converted to a byte vector.
45    data: Vec<Vec<u8>>,
46}
47
48impl ActionData {
49    /// Creates a new [ActionData] mapping the `key` to the `data`.
50    pub fn new<T: ToBytes>(key: &str, data: T) -> Self {
51        ActionData {
52            key: key.to_owned(),
53            data: data.to_bytes(),
54        }
55    }
56
57    /// Returns the `key` mapped to the `data`.
58    pub fn get_key(&self) -> &str {
59        &self.key
60    }
61
62    /// Returns the `data` associated with the `key`.
63    pub fn get_data(&self) -> &Vec<u8> {
64        &self.data
65    }
66
67    /// Returns the data as an [u32].
68    pub fn as_u32(&self) -> u32 {
69        self.get_data().to_u32()
70    }
71}
72
73impl ActionDataRepeated {
74    /// Creates a new [ActionDataRepeated] mapping the `key` to the `data`.
75    pub fn new<T: ToBytes>(key: &str, data: Vec<T>) -> Self {
76        ActionDataRepeated {
77            key: key.to_owned(),
78            data: data.into_iter().map(|d| d.to_bytes()).collect(),
79        }
80    }
81
82    /// Returns the `key` mapped to the `data`.
83    pub fn get_key(&self) -> &str {
84        &self.key
85    }
86
87    /// Returns the `data` associated with the `key`.
88    pub fn get_data(&self) -> &Vec<Vec<u8>> {
89        &self.data
90    }
91}
